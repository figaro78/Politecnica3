epa_http <- epa_http %>%
mutate(url_length = nchar(url))


datos_cluster <- epa_http %>%
select(status, bytes, url_length) %>%
mutate(bytes = ifelse(is.na(bytes), 0,bytes))

set.seed(123)
k2 <- kmeans(datos_cluster, centers = 2, nstart =10)


k3 <- kmeans(datos_cluster, centers = 3, nstart = 10)

datos_cluster$cluster_k2 <- k2$cluster
datos_cluster$cluster_k3 <- k3$cluster

table(datos_cluster$clusyer_k2)
table(datos_cluster$cluster_k3)


datos_cluster_scaled <- scale(datos_cluster[, c("status", "bytes","url_length")])

k2 <- kmeans(datos_cluster_scaled, centers =2, nstart = 10)
k3 <- kmeans(datos_cluster_scaled, centers =3, nstart = 10)

table(k2$cluster)
table(k3$cluster)
